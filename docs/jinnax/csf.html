 28def get_base_config():
 29    """
 30    Base config file for training PINN for CSF in jaxpi
 31
 32    Returns
 33    -------
 34    ml_collections config dictionary
 35    """
 36    #Get the default hyperparameter configuration.
 37    config = ml_collections.ConfigDict()
 38
 39    # Weights & Biases
 40    config.wandb = wandb = ml_collections.ConfigDict()

 42
 43    # Arch
 44    config.arch = arch = ml_collections.ConfigDict()
 45    arch.arch_name = "ModifiedMlp"
 46    arch.num_layers = 4
 47    arch.hidden_dim = 256
 48    arch.out_dim = 2
 49    arch.activation = "tanh"
 50    arch.reparam = ml_collections.ConfigDict(
 51        {"type": "weight_fact", "mean": 1.0, "stddev": 0.1}
 52    )
 53
 54    # Optim
 55    config.optim = optim = ml_collections.ConfigDict()
 56    optim.optimizer = "Adam"
 57    optim.beta1 = 0.9
 58    optim.beta2 = 0.999
 59    optim.eps = 1e-8
 60    optim.learning_rate = 1e-3
 61    optim.decay_rate = 0.9
 62    optim.decay_steps = 2000
 63    optim.grad_accum_steps = 0
 64    optim.warmup_steps = 0
 65    optim.grad_accum_steps = 0
 66
 67    # Training
 68    config.training = training = ml_collections.ConfigDict()
 69    training.batch_size_per_device = 4096
 70
 71    # Weighting
 72    config.weighting = weighting = ml_collections.ConfigDict()
 73    weighting.scheme = "grad_norm"
 74    weighting.momentum = 0.9
 75    weighting.update_every_steps = 1000
 76
 77    weighting.use_causal = True
 78    weighting.causal_tol = 1.0
 79    weighting.num_chunks = 16
 80    optim.staircase = False
 81
 82    # Logging
 83    config.logging = logging = ml_collections.ConfigDict()
 84    logging.log_every_steps = 1000
 85    logging.log_errors = True
 86    logging.log_losses = True
 87    logging.log_weights = True
 88    logging.log_preds = False
 89    logging.log_grads = False
 90    logging.log_ntk = False
 91

104def demo_time_CSF(data,type = 'DN',radius = None,file_name_save = 'result_pinn_CSF_demo',title = '',framerate = 10,ffmpeg = 'ffmpeg'):
105    """
106    Demo video with the time evolution of a CSF in a circle
107    ----------
108    Parameters
109    ----------
110    data : jax.array
111
112        Data with the predicted values
113

115
116        Type of problem

148    ylo = ylo - 0.1*jnp.abs(ylo)

186def train_csf(config: ml_collections.ConfigDict):
187    """
188    Train PINN for CSF in jaxpi
189    ----------
190    Parameters
191    ----------
192    config : ml_collections.ConfigDict
193
194        Dictionary for training PINN in jaxpi
195
196    uninitial : function
197
198        Function that computes the initial condition
199
200
201    Returns
202    -------
203    model, log_dict
204    """
205    if config.save_wandb:
206        wandb_config = config.wandb
207        wandb.init(project = wandb_config.project, name = wandb_config.name)
208
209    # Define the time and space domain
210    dom = jnp.array([[config.tl, config.tu], [config.xl, config.xu]])
211
212    # Initialize the residual sampler
213    res_sampler = iter(UniformSampler(dom, config.training.batch_size_per_device))
214
215    # Initialize the model
216    if config.type_csf == 'DN':
217        model = class_csf.DN_csf(config)
218    elif config.type_csf == 'NN':
219        model = class_csf.NN_csf(config)
220    elif config.type_csf == 'DD':
221        model = class_csf.DD_csf(config)
222    elif config.type_csf == 'closed':
223        model = class_csf.closed_csf(config)
224
225    # Logger
226    logger = Logger()
227
228    # Initialize evaluator
229    key = jax.random.split(jax.random.PRNGKey(config.seed),4)
230    x0_test = jax.random.uniform(key = jax.random.PRNGKey(key[0,0]),minval = config.xl,maxval = config.xu,shape = (config.N0,1))
231    u1_0_test,u2_0_test = config.uinitial(x0_test)
232    tb_test = jax.random.uniform(key = jax.random.PRNGKey(key[1,0]),minval = config.tl,maxval = config.tu,shape = (config.Nb,1))
233    xc_test = jax.random.uniform(key = jax.random.PRNGKey(key[2,0]),minval = config.xl,maxval = config.xu,shape = (config.Nc ** 2,1))
234    tc_test = jax.random.uniform(key = jax.random.PRNGKey(key[3,0]),minval = config.tl,maxval = config.tu,shape = (config.Nc ** 2,1))
235    if config.type_csf == 'DN':
236        evaluator = class_csf.DN_csf_Evaluator(config, model, x0_test, tb_test, xc_test, tc_test, u1_0_test, u2_0_test)
237    elif config.type_csf == 'NN':
238        evaluator = class_csf.NN_csf_Evaluator(config, model, x0_test, tb_test, xc_test, tc_test, u1_0_test, u2_0_test)
239    elif config.type_csf == 'DD':
240        evaluator = class_csf.DD_csf_Evaluator(config, model, x0_test, tb_test, xc_test, tc_test, u1_0_test, u2_0_test)
241    elif config.type_csf == 'closed':
242        evaluator = class_csf.closed_csf_Evaluator(config, model, x0_test, tb_test, xc_test, tc_test, u1_0_test, u2_0_test)
243
244    # jit warm up
245    print("Training CSF...")
246    start_time = time.time()
247    t0 = start_time
248    for step in range(config.training.max_steps):
249        batch = next(res_sampler)
250        model.state = model.step(model.state, batch)
251
252        # Update weights if necessary
253        if config.weighting.scheme in ["grad_norm", "ntk"]:
254            if step % config.weighting.update_every_steps == 0:
255                model.state = model.update_weights(model.state, batch)
256
257        # Log training metrics, only use host 0 to record results
258        if jax.process_index() == 0:
259            if step % config.logging.log_every_steps == 0:
260                # Get the first replica of the state and batch
261                state = jax.device_get(tree_map(lambda x: x[0], model.state))
262                batch = jax.device_get(tree_map(lambda x: x[0], batch))
263                log_dict = evaluator(state, batch)
264                if config.save_wandb:
265                    wandb.log(log_dict, step)
266                end_time = time.time()
267                logger.log_iter(step, start_time, end_time, log_dict)
268                start_time = end_time
269
270        # Saving
271        if config.saving.save_every_steps is not None:
272            if (step + 1) % config.saving.save_every_steps == 0 or (
273                step + 1
274            ) == config.training.max_steps:
275                ckpt_path = os.path.join(os.getcwd(),"ckpt",config.wandb.name)
276                save_checkpoint(model.state, ckpt_path, keep=config.saving.num_keep_ckpts)
277                if config.type_csf == 'DN':
278                    if log_dict['res1_test'] < config.res_tol and log_dict['res2_test'] < config.res_tol and log_dict['ic_rel_test'] < config.ic_tol and log_dict['ld_test'] < config.dn_tol and log_dict['rd_test'] < config.dn_tol and log_dict['ln_test'] < config.dn_tol:
279                        break
280                elif config.type_csf == 'NN':
281                    if log_dict['res1_test'] < config.res_tol and log_dict['res2_test'] < config.res_tol and log_dict['ic_rel_test'] < config.ic_tol and log_dict['ld_test'] < config.dn_tol and log_dict['rd_test'] < config.dn_tol and log_dict['ln_test'] < config.dn_tol and log_dict['rn_test'] < config.dn_tol:
282                        break
283                elif config.type_csf == 'DD':
284                    if log_dict['res1_test'] < config.res_tol and log_dict['res2_test'] < config.res_tol and log_dict['ic_rel_test'] < config.ic_tol and log_dict['ld_test'] < config.dn_tol and log_dict['rd_test'] < config.dn_tol:
285                        break
286                elif config.type_csf == 'closed':
287                    if log_dict['res1_test'] < config.res_tol and log_dict['res2_test'] < config.res_tol and log_dict['ic_rel_test'] < config.ic_tol:
288                        break
289
290    #Run summary
291    log_dict['total_time'] = time.time() - t0
292    log_dict['epochs'] = step + 1
293
294    return model, log_dict

296def evaluate(config: ml_collections.ConfigDict):
297    """
298    Evaluate PINN for CSF trained in jaxpi
299    ----------
300    Parameters
301    ----------
302    config : ml_collections.ConfigDict
303
304        Dictionary for training PINN in jaxpi
305
306    uninitial : function
307
308        Function that computes the initial condition
309
310
311    Returns
312    -------
313    predicted values
314    """
315    # Initialize the model
316    if config.type_csf == 'DN':
317        model = class_csf.DN_csf(config)
318    elif config.type_csf == 'NN':
319        model = class_csf.NN_csf(config)
320    elif config.type_csf == 'DD':
321        model = class_csf.DD_csf(config)
322    elif config.type_csf == 'closed':
323        model = class_csf.closed_csf(config)
324
325    # Restore the checkpoint
326    ckpt_path = os.path.join(
327        os.getcwd(), "ckpt", config.wandb.name,
328    )
329    model.state = restore_checkpoint(model.state, ckpt_path)
330    params = model.state.params
331
332    #Collocation data
333    tx = jnp.array([[t,x] for t in jnp.linspace(config.tl, config.tu, config.Nt) for x in jnp.linspace(config.xl, config.xu, config.Nc)])
334
335    #Predict
336    u1_pred = model.u1_pred_fn(params, tx[:,0], tx[:,1])
337    u2_pred = model.u2_pred_fn(params, tx[:,0], tx[:,1])
338
339    #Save
340    pred = jnp.append(tx,jnp.append(u1_pred.reshape((u1_pred.shape[0],1)),u2_pred.reshape((u2_pred.shape[0],1)),1),1)
341    jnp.save(config.wandb.project + '_' + config.wandb.name + '.npy',pred)
342
343    return pred

445    config.Nb = Nb

447    config.dn_tol = dn_tol

451        config.weighting.init_weights = ml_collections.ConfigDict(

455            'rd': 1.0,

280                        break
281                elif config.type_csf == 'NN':

286                        break
287                elif config.type_csf == 'closed':

289                        break

292    log_dict['total_time'] = time.time() - t0
